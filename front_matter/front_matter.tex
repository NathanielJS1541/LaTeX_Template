% Define the appearance and structure of the front matter for the start of the
% document.

% Generate the title.
\maketitle

% Document revision. In an article, this will appear on the same page as the
% title, underneath the \date field.
\input{front_matter/document_revision}

% Generate the table of contents on a new page. Without \clearpage, this will
% appear on the same page as the title.
\clearpage
\tableofcontents

% Print the glossary on a new page after the ToC, before the main document
% content. Adding the new page here helps to space out the front matter a bit.
\clearpage
\printglossaries
